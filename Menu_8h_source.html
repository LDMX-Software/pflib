262void Menu<TargetType>::steer(TargetType* p_target) {
263 const Line* theMatch = 0;
264 do {
265 printf("\n");
266 if (render_func_ != 0) {
267 this->render_func_(p_target);
268 }
269 if (!quiet_batch_mode)
270 for (size_t i = 0; i < lines_.size(); i++) {
271 printf(" %-12s %s\n", lines_[i].name(), lines_[i].desc());
272 }
273 std::string request = readline(" > ");
274 theMatch = 0;
275 // check for a unique match...
276 int nmatch = 0;
277 for (size_t i = 0; i < lines_.size(); i++)
278 if (strncasecmp(request.c_str(), lines_[i].name(), request.length()) == 0) {
279 theMatch = &(lines_[i]);
280 nmatch++;
281 }
282 if (nmatch > 1) theMatch = 0;
283 // ok
284 if (theMatch == 0)
285 printf(" Command '%s' not understood.\n\n", request.c_str());
286 else {
287 add_to_history(theMatch->name());
288 theMatch->execute(p_target);
289 }
290 } while (theMatch == 0 or not theMatch->empty());
291}
