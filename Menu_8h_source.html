350void Menu<TargetType>::steer(TargetType* p_target) const {
351 this->cmd_options_ = this->command_options(); // we are the captain now
352 const Line* theMatch = 0;
353 do {
354 printf("\n");
355 if (render_func_ != 0) {
356 this->render_func_(p_target);
357 }
358 // if cmd text queue is empty, then print menu for interactive user
359 if (this->cmdTextQueue_.empty())
360 for (size_t i = 0; i < lines_.size(); i++) {
361 printf(" %-12s %s\n", lines_[i].name(), lines_[i].desc());
362 }
363 std::string request = readline_cmd();
364 theMatch = 0;
365 // check for a unique match...
366 int nmatch = 0;
367 for (size_t i = 0; i < lines_.size(); i++)
368 if (strcasecmp(request.c_str(), lines_[i].name()) == 0) {
369 theMatch = &(lines_[i]);
370 nmatch++;
371 }
372 if (nmatch > 1) theMatch = 0;
373 // ok
374 if (theMatch == 0)
375 printf(" Command '%s' not understood.\n\n", request.c_str());
376 else {
377 add_to_history(theMatch->name());
378 if (theMatch->execute(p_target)) {
379 // resume control when the chosen line was a submenu
380 this->cmd_options_ = this->command_options();
381 }
382 }
383 } while (theMatch == 0 or not theMatch->empty());
384}
