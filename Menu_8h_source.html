271void Menu<TargetType>::steer(TargetType* p_target) const {
272 const Line* theMatch = 0;
273 do {
274 printf("\n");
275 if (render_func_ != 0) {
276 this->render_func_(p_target);
277 }
278 if (!quiet_batch_mode)
279 for (size_t i = 0; i < lines_.size(); i++) {
280 printf(" %-12s %s\n", lines_[i].name(), lines_[i].desc());
281 }
282 std::string request = readline(" > ");
283 theMatch = 0;
284 // check for a unique match...
285 int nmatch = 0;
286 for (size_t i = 0; i < lines_.size(); i++)
287 if (strncasecmp(request.c_str(), lines_[i].name(), request.length()) == 0) {
288 theMatch = &(lines_[i]);
289 nmatch++;
290 }
291 if (nmatch > 1) theMatch = 0;
292 // ok
293 if (theMatch == 0)
294 printf(" Command '%s' not understood.\n\n", request.c_str());
295 else {
296 add_to_history(theMatch->name());
297 theMatch->execute(p_target);
298 }
299 } while (theMatch != 0 and not theMatch->empty());
